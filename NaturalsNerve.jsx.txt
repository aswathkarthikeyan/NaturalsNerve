const COST_PER_STYLIST = 8000;

function getBestAllocation(budget) {
  const sorted = [...MODULES].sort((a, b) => b.revLift - a.revLift);
  let rem = budget, totalRet = 0, allocs = [];
  for (const m of sorted) {
    if (rem <= 0) break;
    const n = Math.min(Math.floor(rem / COST_PER_STYLIST), m.stylists);
    if (!n) continue;
    rem -= n * COST_PER_STYLIST;
    totalRet += n * m.revLift;
    allocs.push({ ...m, n, spent: n * COST_PER_STYLIST, ret: n * m.revLift });
  }
  return { allocs, totalRet, budgetUsed: budget - rem };
}
